function solveQuadratic() { let a = parseFloat(document.getElementById('a').value); let b = parseFloat(document.getElementById('b').value); let c = parseFloat(document.getElementById('c').value); let discriminant = b * b - 4 * a * c; let solutionArea = document.getElementById('solutionArea'); if (discriminant > 0) { let x1 = (-b + Math.sqrt(discriminant)) / (2 * a); let x2 = (-b - Math.sqrt(discriminant)) / (2 * a); solutionArea.innerHTML = `x1 = ${x1}, x2 = ${x2}`; } else if (discriminant === 0) { let x = -b / (2 * a); solutionArea.innerHTML = `x = ${x}`; } else { let realPart = -b / (2 * a); let imaginaryPart = Math.sqrt(-discriminant) / (2 * a); solutionArea.innerHTML = `x1 = ${realPart}+${imaginaryPart}i, x2 = ${realPart}-${imaginaryPart}i`; } } form { display: flex; gap: 10px; margin-bottom: 10px; } input { padding: 5px; } button { padding: 5px 10px; } #solutionArea { border: 1px solid #ccc; padding: 10px; }
